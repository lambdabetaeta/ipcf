        \ctxt{\Delta}{\cdot} \vdash M : A
          \qquad
        \ctxt{\Delta, u : A}{\Gamma} \vdash N : C
          \justifies
        \ctxt{\Delta}{\Gamma}
            \vdash \letbox{u}{\ibox{M}}{N} = N[M/x] : C
          \using
        {(\Box\beta)}
      \end{prooftree}
